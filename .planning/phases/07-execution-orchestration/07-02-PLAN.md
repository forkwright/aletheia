    ```typescript
      this.store.updateProjectState(projectId, transition("executing", "VERIFY"));
      eventBus.emit("planning:phase-complete", { projectId, nousId, sessionId, fromState: "executing", toState: "verifying" });

      this.store.updateProjectState(projectId, transition("executing", "BLOCK"));
      log.info(`Execution paused for project ${projectId}`);

      this.store.updateProjectState(projectId, transition("blocked", "RESUME"));
      eventBus.emit("planning:phase-started", { projectId, nousId, sessionId, fromState: "blocked", toState: "executing" });
      log.info(`Execution resumed for project ${projectId}`);
      return "Execution resumed.";
    }
    ```

    Create `infrastructure/runtime/src/dianoia/execution-tool.ts` following the roadmap-tool.ts action-dispatch pattern exactly.

    **Tool name:** `plan_execute`
    **Description:** "Execute, pause, resume, retry, skip, or abandon a Dianoia phase execution. Use action=start to begin wave-based parallel execution, action=status to check progress."

    **Actions and their behavior:**

    | Action | Required Input | Effect |
    |--------|---------------|--------|
    | `start` | projectId, phaseId | Call executionOrchestrator.executePhase(projectId, phaseId, context). On completion, call planningOrchestrator.advanceToVerification(). |
    | `pause` | projectId | Call planningOrchestrator.pauseExecution(projectId). Returns "Execution will pause after the current wave completes." |
    | `resume` | projectId, phaseId | Call planningOrchestrator.resumeExecution(). Then call executionOrchestrator.executePhase() to continue from resume wave. |
    | `retry` | projectId, phaseId, planId | Delete spawn records for planId (update to 'pending' is simpler than delete — set status='pending', clear error/completedAt). Then re-run executePhase from that wave. |
    | `skip` | projectId, phaseId, planId | Find spawn record for planId, update status='skipped'. Apply directDependents cascade-skip via store.updateSpawnRecord(). Return "Plan skipped." |
    | `abandon` | projectId | Call planningOrchestrator.abandon(projectId). Returns "Phase execution abandoned." |
    | `status` | projectId, phaseId (optional) | Call executionOrchestrator.getExecutionSnapshot(projectId). Return JSON snapshot. |

    **Implementation pattern (from roadmap-tool.ts):**

    ```typescript
    // execution-tool.ts
    import { z } from "zod";
    import type { DianoiaOrchestrator } from "./orchestrator.js";
    import type { ExecutionOrchestrator } from "./execution.js";
    import type { ToolContext, ToolHandler } from "../organon/registry.js";

    const ExecuteInput = z.object({
      action: z.enum(["start", "pause", "resume", "retry", "skip", "abandon", "status"]),
      projectId: z.string(),
      phaseId: z.string().optional(),
      planId: z.string().optional(),
      nousId: z.string().optional(),
      sessionId: z.string().optional(),
    });

    export function createPlanExecuteTool(
      planningOrchestrator: DianoiaOrchestrator,
      executionOrchestrator: ExecutionOrchestrator,
    ): ToolHandler {
      return {
        name: "plan_execute",
        description: "Execute, pause, resume, retry, skip, or abandon a Dianoia phase execution. Use action=start to begin wave-based parallel execution, action=status to check progress.",
        inputSchema: ExecuteInput,
        execute(input: unknown, context: ToolContext): Promise<string> {
          const parsed = ExecuteInput.parse(input);
          return handleAction(parsed, planningOrchestrator, executionOrchestrator, context);
        },
      };
    }

    async function handleAction(
      input: z.infer<typeof ExecuteInput>,
      planningOrchestrator: DianoiaOrchestrator,
      executionOrchestrator: ExecutionOrchestrator,
      context: ToolContext,
    ): Promise<string> {
      const { action, projectId, phaseId, planId, nousId = "", sessionId = "" } = input;
      switch (action) {
        case "start": {
          if (!phaseId) throw new Error("phaseId required for start");
          const result = await executionOrchestrator.executePhase(projectId, phaseId, context);
          if (result.failed === 0) {
            return planningOrchestrator.advanceToVerification(projectId, nousId, sessionId);
          }
          return JSON.stringify({ ...result, message: "Execution complete with failures. Use action=retry or action=skip to recover." });
        }
        case "pause":
          planningOrchestrator.pauseExecution(projectId);
          return "Execution will pause after the current wave completes.";
        case "resume": {
          if (!phaseId) throw new Error("phaseId required for resume");
          const msg = planningOrchestrator.resumeExecution(projectId, nousId, sessionId);
          await executionOrchestrator.executePhase(projectId, phaseId, context);
          return msg;
        }
        case "retry": {
          if (!phaseId || !planId) throw new Error("phaseId and planId required for retry");
          // Reset spawn record to pending so executePhase re-dispatches it
          const snapshot = executionOrchestrator.getExecutionSnapshot(projectId);
          const planEntry = snapshot.plans.find(p => p.phaseId === planId);
          if (!planEntry) throw new Error(`Plan ${planId} not found in execution snapshot`);
          // ExecutionOrchestrator handles resume-from-wave; marking as pending lets it re-dispatch
          // (findResumeWave skips 'done'/'skipped' only — 'pending' re-enters the wave)
          await executionOrchestrator.executePhase(projectId, phaseId, context);
          return `Retrying plan ${planId} from beginning.`;
        }
        case "skip": {
          if (!phaseId || !planId) throw new Error("phaseId and planId required for skip");
          const snapshot = executionOrchestrator.getExecutionSnapshot(projectId);
          const planEntry = snapshot.plans.find(p => p.phaseId === planId);
          if (!planEntry) throw new Error(`Plan ${planId} not found`);
          return `Plan ${planId} skipped. Partial commits left in place.`;
        }
        case "abandon":
          planningOrchestrator.abandon(projectId);
          return "Phase execution abandoned.";
        case "status": {
          const snapshot = executionOrchestrator.getExecutionSnapshot(projectId);
          return JSON.stringify(snapshot, null, 2);
        }
      }
    }
    ```

    **Critical rules from research:**
    - Do NOT use `async` keyword on the outer `execute()` method — wrap in `Promise.resolve()` or use direct return of async function to satisfy oxlint require-await. The pattern above (returning `handleAction()` which is async) is correct.
    - Do NOT re-implement concurrency — sessions_dispatch handles that inside executePhase().
    - retry action: the simplest correct implementation is to call executePhase() again (findResumeWave will detect the failed record and re-dispatch). If the spawn record needs resetting, use store.updateSpawnRecord() to set status back to 'pending'.
